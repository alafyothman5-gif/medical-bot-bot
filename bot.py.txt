class PostgresPersistence(BasePersistence):
    """Stores bot persistence data in PostgreSQL via DATABASE_URL.

    All session data is held durably in the database so it survives container
    restarts and can be inspected or wiped per-user with plain SQL.

    Writes use psycopg2 (synchronous) inside asyncio.to_thread() to avoid
    blocking the event loop.
    """

    def __init__(self, dsn: str) -> None:
        super().__init__()
        self._dsn = dsn
        self._user_data: DefaultDict[int, Dict] = defaultdict(dict)
        self._chat_data: DefaultDict[int, Dict] = defaultdict(dict)
        self._bot_data: Dict = {}
        self._conversations: Dict[str, Dict[Tuple, object]] = {}
        self._loaded = False

    # ── low-level helpers (run in thread executor) ────────────────────────

    def _connect(self) -> psycopg2.extensions.connection:
        conn = psycopg2.connect(self._dsn)
        with conn.cursor() as cur:
            cur.execute(
                """
                CREATE TABLE IF NOT EXISTS bot_persistence (
                    store TEXT NOT NULL,
                    key   TEXT NOT NULL,
                    value TEXT NOT NULL,
                    PRIMARY KEY (store, key)
                )
                """
            )
        conn.commit()
        return conn

    def _load_all_sync(self) -> None:
        if self._loaded:
            return
        conn = self._connect()
        try:
            with conn.cursor() as cur:
                cur.execute("SELECT store, key, value FROM bot_persistence")
                for store, key, value in cur.fetchall():
                    data = json.loads(value)
                    if store == "user_data":
                        self._user_data[int(key)] = data
                    elif store == "chat_data":
                        self._chat_data[int(key)] = data
                    elif store == "bot_data":
                        self._bot_data = data
                    elif store == "conversations":
                        self._conversations[key] = {
                            tuple(json.loads(k)): v for k, v in data.items()
                        }
        finally:
            conn.close()
        self._loaded = True

    def _save_row_sync(self, store: str, key: str, value: object) -> None:
        conn = self._connect()
        try:
            with conn.cursor() as cur:
                cur.execute(
                    """
                    INSERT INTO bot_persistence (store, key, value)
                    VALUES (%s, %s, %s)
                    ON CONFLICT (store, key) DO UPDATE SET value = EXCLUDED.value
                    """,
                    (store, key, json.dumps(value, default=str)),
                )
            conn.commit()
        finally:
            conn.close()

    def _delete_row_sync(self, store: str, key: str) -> None:
        conn = self._connect()
        try:
            with conn.cursor() as cur:
                cur.execute(
                    "DELETE FROM bot_persistence WHERE store = %s AND key = %s",
                    (store, key),
                )
            conn.commit()
        finally:
            conn.close()

    # ── async wrappers ────────────────────────────────────────────────────

    async def _load_all(self) -> None:
        await asyncio.to_thread(self._load_all_sync)

    async def _save_row(self, store: str, key: str, value: object) -> None:
        await asyncio.to_thread(self._save_row_sync, store, key, value)

    async def _delete_row(self, store: str, key: str) -> None:
        await asyncio.to_thread(self._delete_row_sync, store, key)

    # ── BasePersistence interface ─────────────────────────────────────────

    async def get_user_data(self) -> DefaultDict[int, Dict]:
        await self._load_all()
        return self._user_data

    async def get_chat_data(self) -> DefaultDict[int, Dict]:
        await self._load_all()
        return self._chat_data

    async def get_bot_data(self) -> Dict:
        await self._load_all()
        return self._bot_data

    async def get_callback_data(self) -> Optional[Any]:
        return None

    async def get_conversations(self, name: str) -> Dict[Tuple, Any]:
        await self._load_all()
        return self._conversations.get(name, {})

    async def update_user_data(self, user_id: int, data: Dict) -> None:
        self._user_data[user_id] = data
        await self._save_row("user_data", str(user_id), data)

    async def update_chat_data(self, chat_id: int, data: Dict) -> None:
        self._chat_data[chat_id] = data
        await self._save_row("chat_data", str(chat_id), data)

    async def update_bot_data(self, data: Dict) -> None:
        self._bot_data = data
        await self._save_row("bot_data", "__bot__", data)

    async def update_callback_data(self, data: Any) -> None:
        pass

    async def update_conversation(
        self, name: str, key: Tuple, new_state: Optional[object]
    ) -> None:
        if name not in self._conversations:
            self._conversations[name] = {}
        self._conversations[name][key] = new_state
        serialised = {
            json.dumps(list(k)): v for k, v in self._conversations[name].items()
        }
        await self._save_row("conversations", name, serialised)

    async def drop_user_data(self, user_id: int) -> None:
        self._user_data.pop(user_id, None)
        await self._delete_row("user_data", str(user_id))

    async def drop_chat_data(self, chat_id: int) -> None:
        self._chat_data.pop(chat_id, None)
        await self._delete_row("chat_data", str(chat_id))

    async def flush(self) -> None:
        pass  # every write is committed immediately

    async def refresh_user_data(self, user_id: int, user_data: Dict) -> None:
        pass

    async def refresh_chat_data(self, chat_id: int, chat_data: Dict) -> None:
        pass

    async def refresh_bot_data(self, bot_data: Dict) -> None:
        pass


def _migrate_pickle_to_postgres(pkl_path: str, dsn: str) -> None:
    """One-shot migration: import data from an existing pickle file into Postgres.

    Enforces idempotency in two ways:
    1. Queries bot_persistence before importing; skips if any rows already exist
       (meaning a previous migration or normal bot operation has written data).
    2. Renames the pickle file to quiz_sessions.pkl.migrated after a successful
       import so the file can no longer trigger this path on restart.

    On any error the migration is skipped and the bot starts with whatever state
    is already in the database.
    """
    _log = logging.getLogger(__name__)
    try:
        persistence = PostgresPersistence(dsn=dsn)

        # Ensure the table exists and check for existing rows.
        conn = persistence._connect()
        try:
            with conn.cursor() as cur:
                cur.execute("SELECT COUNT(*) FROM bot_persistence")
                existing_rows = cur.fetchone()[0]
        finally:
            conn.close()

        if existing_rows > 0:
            _log.info(
                "Pickle migration skipped: bot_persistence already has %d row(s).",
                existing_rows,
            )
            return

        with open(pkl_path, "rb") as fh:
            data = pickle.load(fh)

        user_data = data.get("user_data") or {}
        u_ok, u_fail = 0, 0
        for uid, udata in user_data.items():
            try:
                persistence._save_row_sync("user_data", str(uid), udata)
                u_ok += 1
            except Exception as row_err:
                _log.warning("Migration: failed to import user_data uid=%s: %s", uid, row_err)
                u_fail += 1

        chat_data = data.get("chat_data") or {}
        c_ok, c_fail = 0, 0
        for cid, cdata in chat_data.items():
            try:
                persistence._save_row_sync("chat_data", str(cid), cdata)
                c_ok += 1
            except Exception as row_err:
                _log.warning("Migration: failed to import chat_data cid=%s: %s", cid, row_err)
                c_fail += 1

        bot_data = data.get("bot_data") or {}
        if bot_data:
            try:
                persistence._save_row_sync("bot_data", "__bot__", bot_data)
            except Exception as row_err:
                _log.warning("Migration: failed to import bot_data: %s", row_err)

        _log.info(
            "Migrated pickle to Postgres: users %d ok / %d failed, chats %d ok / %d failed",
            u_ok, u_fail, c_ok, c_fail,
        )

        # Rename the pickle so it can't trigger migration again on next restart.
        try:
            os.rename(pkl_path, pkl_path + ".migrated")
        except OSError as rename_err:
            _log.warning("Could not rename pickle after migration: %s", rename_err)

    except Exception as exc:
        _log.warning("Pickle migration skipped (%s). Starting with empty state.", exc)

# RESULT CACHE: deduplicate identical uploads across users
# ─────────────────────────────────────────────────────────────
_RESULT_CACHE_SIZE = int(os.environ.get("RESULT_CACHE_SIZE", "512"))
_RESULT_CACHE: "OrderedDict[tuple, Any]" = OrderedDict()
_RESULT_CACHE_LOCK = threading.Lock()


def _cache_get(key: tuple) -> Any:
    with _RESULT_CACHE_LOCK:
        val = _RESULT_CACHE.get(key)
        if val is None:
            return None
        _RESULT_CACHE.move_to_end(key)
    return copy.deepcopy(val)


def _cache_put(key: tuple, value: Any) -> None:
    snapshot = copy.deepcopy(value)
    with _RESULT_CACHE_LOCK:
        if key in _RESULT_CACHE:
            _RESULT_CACHE.move_to_end(key)
            _RESULT_CACHE[key] = snapshot
            return
        _RESULT_CACHE[key] = snapshot
        while len(_RESULT_CACHE) > _RESULT_CACHE_SIZE:
            _RESULT_CACHE.popitem(last=False)


def _md5_bytes(data: bytes) -> str:
    return hashlib.md5(data).hexdigest()

def get_lang(user_data: dict) -> str:
    lang = user_data.get("lang", "en")
    return lang if lang in SUPPORTED_LANGS else "en"


def t(key: str, lang: str = "en", **kwargs) -> str:
    bundle = TRANSLATIONS.get(lang) or TRANSLATIONS["en"]
    text = bundle.get(key) or TRANSLATIONS["en"].get(key, key)
    if kwargs:
        try:
            return text.format(**kwargs)
        except Exception as e:
            logger.warning(f"t() format error for key={key}: {e}")
            return text
    return text


def diff_label(difficulty: str, lang: str) -> str:
    return t(f"diff_{difficulty}_label", lang)

_SCANNED_PDF_MIN_CHARS = 50
_SCANNED_PDF_OCR_PAGES = 5
_SCANNED_PDF_RENDER_DPI = 200


def _render_scanned_pdf_pages(pdf_bytes: bytes) -> list[bytes]:
    import fitz
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page_count = min(len(doc), _SCANNED_PDF_OCR_PAGES)
        if page_count == 0:
            return []

        zoom = _SCANNED_PDF_RENDER_DPI / 72.0
        matrix = fitz.Matrix(zoom, zoom)

        out: list[bytes] = []
        for idx in range(page_count):
            try:
                pixmap = doc.load_page(idx).get_pixmap(matrix=matrix, alpha=False)
                out.append(pixmap.tobytes("jpeg"))
            except Exception as e:
                logger.warning(
                    "_render_scanned_pdf_pages: page %d render failed: %s — skipping",
                    idx + 1, e,
                )
                out.append(b"")
        return out
    finally:
        doc.close()


def _ocr_scanned_pdf(pdf_bytes: bytes) -> str:
    pages_jpeg = _render_scanned_pdf_pages(pdf_bytes)
    if not pages_jpeg:
        return ""
    chunks: list[str] = []
    for idx, jpeg in enumerate(pages_jpeg):
        if not jpeg:
            continue
        try:
            page_text = extract_text_from_image(jpeg, "image/jpeg")
            if page_text:
                chunks.append(page_text)
        except Exception as e:
            logger.warning(
                "_ocr_scanned_pdf: page %d Vision failed: %s — skipping",
                idx + 1, e,
            )
    return "\n\n".join(chunks)


async def _ocr_scanned_pdf_async(pdf_bytes: bytes) -> str:
    pages_jpeg = await asyncio.to_thread(_render_scanned_pdf_pages, pdf_bytes)
    if not pages_jpeg:
        return ""

    async def _ocr_one(idx: int, jpeg: bytes) -> tuple[int, str]:
        if not jpeg:
            return (idx, "")
        try:
            text = await asyncio.to_thread(
                extract_text_from_image, jpeg, "image/jpeg"
            )
            return (idx, text or "")
        except Exception as e:
            logger.warning(
                "_ocr_scanned_pdf_async: page %d Vision failed: %s — skipping",
                idx + 1, e,
            )
            return (idx, "")

    logger.info(
        "_ocr_scanned_pdf_async: dispatching %d Vision calls in parallel",
        len(pages_jpeg),
    )
    results = await asyncio.gather(
        *[_ocr_one(i, j) for i, j in enumerate(pages_jpeg)]
    )
    results.sort(key=lambda x: x[0])
    return "\n\n".join(text for _, text in results if text)


_MAX_PDF_TEXT_CHARS = 50_000


def _truncate_pdf_text(text: str, source: str) -> str:
    if len(text) <= _MAX_PDF_TEXT_CHARS:
        return text
    logger.info(
        "extract_text_from_pdf_async: %s text=%d chars exceeds cap=%d — "
        "truncating to first %d chars",
        source, len(text), _MAX_PDF_TEXT_CHARS, _MAX_PDF_TEXT_CHARS,
    )
    return text[:_MAX_PDF_TEXT_CHARS]


async def extract_text_from_pdf_async(pdf_bytes: bytes) -> str:
    def _text_layer() -> str:
        reader = PdfReader(io.BytesIO(pdf_bytes))
        pages: list[str] = []
        for page in reader.pages:
            s = page.extract_text()
            if s:
                pages.append(s.strip())
        return "\n\n".join(pages)

    text = await asyncio.to_thread(_text_layer)

    if len(text.strip()) >= _SCANNED_PDF_MIN_CHARS:
        return _truncate_pdf_text(text, "text-layer")

    logger.info(
        "extract_text_from_pdf_async: text layer too short (%d chars) — "
        "parallel Vision OCR on first %d pages",
        len(text.strip()), _SCANNED_PDF_OCR_PAGES,
    )
    try:
        ocr_text = await _ocr_scanned_pdf_async(pdf_bytes)
    except Exception as e:
        logger.error(
            "extract_text_from_pdf_async: parallel OCR failed: %s",
            e, exc_info=True,
        )
        return text

    chosen = ocr_text if ocr_text.strip() else text
    return _truncate_pdf_text(chosen, "ocr-fallback")


def extract_text_from_pdf(pdf_bytes: bytes) -> str:
    reader = PdfReader(io.BytesIO(pdf_bytes))
    pages = []
    for page in reader.pages:
        s = page.extract_text()
        if s:
            pages.append(s.strip())
    text = "\n\n".join(pages)

    if len(text.strip()) >= _SCANNED_PDF_MIN_CHARS:
        return text

    logger.info(
        "extract_text_from_pdf: text layer too short (%d chars) — "
        "falling back to Vision OCR on first %d pages",
        len(text.strip()), _SCANNED_PDF_OCR_PAGES,
    )
    try:
        ocr_text = _ocr_scanned_pdf(pdf_bytes)
    except Exception as e:
        logger.error("extract_text_from_pdf: OCR fallback failed: %s", e)
        return text

    return ocr_text if ocr_text.strip() else text


def extract_text_from_image(image_bytes: bytes, mime_type: str = "image/jpeg") -> str:
    base64_image = base64.b64encode(image_bytes).decode("utf-8")
    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {
                "role": "user",
                "content": [

                            "Extract and transcribe all text from this image perfectly. "
                            "Maintain original language and layout. Do not summarize, "
                            "just give me the raw transcribed text."
                        ),
                    },
                    {
                        "type": "image_url",
                        "image_url": {
                            "url": f"data:image/jpeg;base64,{base64_image}",
                        },
                    },
                ],
            }
        ],
        max_tokens=2000,
        temperature=0.0,
    )
    extracted_text = response.choices[0].message.content
    return extracted_text.strip() if extracted_text else ""


def extract_text_from_docx(docx_bytes: bytes) -> str:
    from docx import Document
    doc = Document(io.BytesIO(docx_bytes))
    parts = [para.text for para in doc.paragraphs if para.text.strip()]
    return "\n\n".join(parts)


def extract_text_from_txt(txt_bytes: bytes) -> str:
    for enc in ("utf-8", "utf-16", "latin-1"):
        try:
            return txt_bytes.decode(enc)
        except UnicodeDecodeError:
            continue
    return txt_bytes.decode("utf-8", errors="replace")

def _split_summary_chunks(text: str, max_len: int = 3800) -> list[str]:
    if not text:
        return []
    if len(text) <= max_len:
        return [text]

    chunks: list[str] = []
    cur: list[str] = []
    cur_len = 0
    inside_fence = False

    def _flush() -> None:
        nonlocal cur, cur_len
        if not cur:
            return
        piece = "\n".join(cur)
        if inside_fence:
            piece = piece + "\n```"
        chunks.append(piece)
        cur = []
        cur_len = 0
        if inside_fence:
            cur.append("```")
            cur_len = len("```")

    for line in text.split("\n"):
        line_len = len(line) + 1

        if line_len > max_len:
            _flush()
            for i in range(0, len(line), max_len):
                chunks.append(line[i:i + max_len])
            continue

        if cur_len + line_len > max_len:
            _flush()

        cur.append(line)
        cur_len += line_len

        if line.lstrip().startswith("```"):
            inside_fence = not inside_fence

    _flush()
    return chunks

def generate_detective_turn(
    context_text: str,
    history: list,
    round_num: int,
    max_rounds: int = DETECTIVE_MAX_ROUNDS,
) -> str:
    context_text = (context_text or "").strip()
    if len(context_text) > 6_000:
        context_text = context_text[:6_000]

    messages = [
        {"role": "system", "content": _detective_system_prompt(round_num, max_rounds)},
        {"role": "user",   "content": (
            "Case anchor (background reference for you — do NOT echo verbatim):\n"
            f"{context_text}"
        )},
    ]
    for m in history:
        if isinstance(m, dict) and m.get("role") in ("user", "assistant") and m.get("content"):
            messages.append({"role": m["role"], "content": m["content"]})

    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        max_completion_tokens=800,
        temperature=0.5,
        messages=messages,
    )
    return response.choices[0].message.content.strip()

def shuffle_question_options(q: dict) -> dict:
    letters = ["A", "B", "C", "D", "E"]
    if "options" not in q or "correct" not in q:
        return q
    correct_letter = str(q["correct"]).strip().upper()
    if correct_letter not in q["options"]:
        return q

    items = [(q["options"][L], L == correct_letter) for L in letters if L in q["options"]]
    if len(items) != 5:
        return q

    random.shuffle(items)

    new_options = {}
    new_correct = correct_letter
    for i, (text, is_correct) in enumerate(items):
        L = letters[i]
        new_options[L] = text
        if is_correct:
            new_correct = L

    q["options"] = new_options
    q["correct"] = new_correct
    return q


def escape_md(text: object) -> str:
    text = "" if text is None else str(text)
    for ch in ("*", "_", "`", "["):
        text = text.replace(ch, f"\{ch}")
    return text


def progress_bar(current: int, total: int, width: int = 12) -> str:
    filled = round(current / total * width) if total > 0 else 0
    bar = "█" * filled + "░" * (width - filled)
    pct = round(current / total * 100) if total > 0 else 0
    return f"{bar}  {pct}%"

def _validate_nickname(raw: str) -> tuple[str, str | None]:
    clean = (raw or "").strip()
    if not clean:
        return "", "nickname_invalid"
    if len(clean) > _NICKNAME_MAX_LEN:
        return "", "nickname_too_long"
    if not _NICKNAME_ALLOWED_RE.match(clean):
        return "", "nickname_invalid"
    return clean, None


def _best_display_name(user) -> tuple[str, str]:
    if user is None:
        return "", ""
    first = (getattr(user, "first_name", None) or "").strip()
    last  = (getattr(user, "last_name", None) or "").strip()
    uname = (getattr(user, "username", None) or "").strip()
    full = (f"{first} {last}".strip()) if (first or last) else ""
    display = full or (f"@{uname}" if uname else "")
    return display, (f"@{uname}" if uname else "")


def _capture_display_name(user_data: dict | None, user) -> None:
    if user_data is None or user is None:

    display, username = _best_display_name(user)
    if display:
        user_data["display_name"] = display
    if username:
        user_data["username"] = username

def _generate_mcq_chunk(
    text: str,
    count: int,
    difficulty: str,
    batch_idx: int,
    total_batches: int,
    style: str | None = None,
) -> list[dict]:
    config = DIFFICULTY_CONFIGS.get(difficulty, DIFFICULTY_CONFIGS["hard"])
    perspective = _BATCH_PERSPECTIVES[batch_idx % len(_BATCH_PERSPECTIVES)]
    system_prompt = config["system_prompt"]
    if style and style in STYLE_MODIFIERS:
        system_prompt = system_prompt + "\n\n" + STYLE_MODIFIERS[style]

    if total_batches > 1:
        user_prompt = (
            f"Generate exactly {count} MCQs at the difficulty level specified in the system prompt.\n"
            f"This is batch {batch_idx + 1} of {total_batches} running in parallel. "
            f"FOCUS ANGLE FOR THIS BATCH: {perspective} "
            "Produce questions DISTINCT from other batches by leaning into this focus angle. "
            "Never duplicate questions.\n"
            "All questions MUST be written in English only. Do not use any other language.\n\n"
            f"SOURCE TEXT:\n{text}"
        )
    else:
        user_prompt = (
            f"Generate exactly {count} MCQs at the difficulty level specified in the system prompt. "
            "If the source text is too short or repetitive, generate as many distinct high-quality questions as possible — "
            "but never duplicate questions.\n"
            "All questions MUST be written in English only. Do not use any other language.\n\n"
            f"SOURCE TEXT:\n{text}"
        )

    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        max_completion_tokens=4096,
        temperature=0.4,
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ],
    )

    raw = response.choices[0].message.content.strip()
    json_match = re.search(r"\[.*\]", raw, re.DOTALL)
    if json_match:
        raw = json_match.group(0)
    parsed = json.loads(raw)
    return [q for q in parsed if isinstance(q, dict)]


def generate_mcqs(
    text: str,
    count: int = 20,
    difficulty: str = "hard",
    style: str | None = None,
) -> list[dict]:
    MAX_CHARS = 40_000
    if len(text) > MAX_CHARS:
        text = text[:MAX_CHARS]

    if count <= _MCQ_BATCH_SIZE:
        questions = _generate_mcq_chunk(text, count, difficulty, 0, 1, style)
        logger.info(
            "generate_mcqs: single-batch path count=%d difficulty=%s style=%s",
            count, difficulty, style or "default",
        )
    else:
        sizes: list[int] = []
        remaining = count
        while remaining > 0:
            sizes.append(min(_MCQ_BATCH_SIZE, remaining))
            remaining -= _MCQ_BATCH_SIZE
        total_batches = len(sizes)

        logger.info(
            "generate_mcqs: chunked path count=%d batches=%d difficulty=%s style=%s",
            count, total_batches, difficulty, style or "default",
        )

        from concurrent.futures import ThreadPoolExecutor, as_completed
        questions: list[dict] = []
        errors: list[Exception] = []
        with ThreadPoolExecutor(max_workers=min(total_batches, _MCQ_MAX_PARALLEL)) as pool:
            futures = [
                pool.submit(_generate_mcq_chunk, text, sz, difficulty, i, total_batches, style)
                for i, sz in enumerate(sizes)
            ]
            for fut in as_completed(futures):
                try:
                    questions.extend(fut.result())
                except Exception as e:
                    errors.append(e)
                    logger.error("generate_mcqs: chunk failed: %s", e, exc_info=True)

        if not questions and errors:
            raise errors[0]

        if len(questions) > count:
            questions = questions[:count]

        logger.info(
            "generate_mcqs: chunked path produced %d questions (requested %d, errors=%d)",
            len(questions), count, len(errors),
        )

    for q in questions:
        if isinstance(q, dict):
            shuffle_question_options(q)

    if len(questions) >= 5:
        dist = {}
        for q in questions:
            c = str(q.get("correct", "")).upper()
            dist[c] = dist.get(c, 0) + 1
        if len(dist) < 2:
            logger.warning(
                "Suspicious correct-letter distribution after shuffle: %s — re-shuffling once.", dist
            )
            for q in questions:
                if isinstance(q, dict):
                    shuffle_question_options(q)
        else:
            logger.info("Correct-letter distribution: %s", dist)

    return questions

def format_review_question(entry: dict, index: int, total: int, lang: str) -> str:

        f"*D.* {escape_md(opts['D'])}\n"

def get_stats(user_data: dict) -> dict:
    if "stats" not in user_data:
        user_data["stats"] = {
            "quizzes_completed": 0,
            "total_correct": 0,
            "total_questions": 0,
            "highest_pct": 0,
            "difficulty_counts": {"easy": 0, "medium": 0, "hard": 0, "nightmare": 0},
        }
    return user_data["stats"]


def update_stats(user_data: dict, score: int, total: int, difficulty: str) -> None:
    stats = get_stats(user_data)
    pct = round(score / total * 100) if total > 0 else 0
    stats["quizzes_completed"] += 1
    stats["total_correct"] += score
    stats["total_questions"] += total
    if pct > stats["highest_pct"]:
        stats["highest_pct"] = pct
    dc = stats.setdefault("difficulty_counts", {})
    dc[difficulty] = dc.get(difficulty, 0) + 1


XP_REWARDS: dict[str, int] = {
    "easy":      10,
    "medium":    20,
    "hard":      30,
    "nightmare": 50,
}

_RANK_THRESHOLDS: list[tuple[int, str]] = [

def _today_utc() -> str:
    return datetime.now(timezone.utc).date().isoformat()


def get_xp(user_data: dict) -> int:
    if not isinstance(user_data.get("xp"), int):
        user_data["xp"] = 0
    return user_data["xp"]


def get_streak(user_data: dict) -> int:
    raw = user_data.get("streak_count", 0)
    try:
        return max(0, int(raw or 0))
    except (TypeError, ValueError):
        return 0


XP_UNLOCK_COST: int = 100


def try_spend_xp(user_data: dict, cost: int) -> bool:
    current = get_xp(user_data)
    if current < cost:
        return False
    user_data["xp"] = current - cost
    return True


async def _flush_user_data(application, user_id) -> None:
    try:
        application.mark_data_for_update_persistence(user_ids={int(user_id)})
        await application.update_persistence()
    except Exception:
        logger.exception("_flush_user_data: failed for user_id=%s", user_id)


_UNLOCK_LOCKS: DefaultDict[int, asyncio.Lock] = defaultdict(asyncio.Lock)
_UNLOCK_LAST_TS: dict[tuple[int, str], float] = {}
_UNLOCK_DEDUPE_WINDOW_SEC: float = 3.0


async def spend_unlock(
    application, user_data: dict, user_id: int, cost: int, action_key: str,
) -> str:
    try:
        uid = int(user_id)
    except (TypeError, ValueError):
        uid = 0

    if _is_admin(uid):
        return "bypass"

    key = (uid, action_key)
    lock = _UNLOCK_LOCKS[uid]
    async with lock:
        now = time.monotonic()
        last = _UNLOCK_LAST_TS.get(key, 0.0)
        if now - last < _UNLOCK_DEDUPE_WINDOW_SEC:
            return "duplicate"
        if not try_spend_xp(user_data, cost):
            return "insufficient"
        _UNLOCK_LAST_TS[key] = now
    await _flush_user_data(application, uid)
    return "ok"


def award_xp(user_data: dict, difficulty: str, correct: bool) -> int:
    if correct:
        delta = XP_REWARDS.get((difficulty or "").lower(), 0)
    else:
        delta = -XP_PENALTY
    current = get_xp(user_data)
    new_total = max(0, current + delta)
    user_data["xp"] = new_total
    return new_total - current


def update_streak(user_data: dict) -> int:
    today_iso = _today_utc()
    last_iso = user_data.get("last_interaction_date")
    streak = int(user_data.get("streak_count", 0) or 0)

    if last_iso == today_iso:
        if streak < 1:
            streak = 1
            user_data["streak_count"] = streak
        return streak

    new_streak = 1
    if isinstance(last_iso, str):
        try:
            last_date = datetime.strptime(last_iso, "%Y-%m-%d").date()
            today_date = datetime.strptime(today_iso, "%Y-%m-%d").date()
            if today_date - last_date == timedelta(days=1) and streak >= 1:
                new_streak = streak + 1
        except (ValueError, TypeError):
            new_streak = 1

    user_data["streak_count"] = new_streak
    user_data["last_interaction_date"] = today_iso
    return new_streak


def get_rank(xp: int, lang: str) -> str:
    rank_key = "rank_novice"
    for threshold, key in _RANK_THRESHOLDS:
        if xp >= threshold:
            rank_key = key
    return t(rank_key, lang)


def streak_suffix(user_data: dict, lang: str) -> str:
    streak = get_streak(user_data)
    if streak <= 0:
        return ""
    return t("welcome_streak", lang, days=streak)


def format_stats_text(user_data: dict, name: str | None = None) -> str:
    lang = get_lang(user_data)
    stats = get_stats(user_data)

    total_q = stats["total_questions"]
    total_c = stats["total_correct"]
    accuracy = round(total_c / total_q * 100) if total_q > 0 else 0
    xp = get_xp(user_data)
    streak = get_streak(user_data)
    rank = get_rank(xp, lang)
    display_name = (name or "—").strip() or "—"

    unit_key = "stats_streak_unit_day" if streak == 1 else "stats_streak_unit_days"

    return (
        f"{t('stats_header', lang)}\n\n"
        f"{t('stats_user',     lang, name=display_name)}\n"
        f"{t('stats_streak',   lang, days=streak, unit=t(unit_key, lang))}\n"
        f"{t('stats_xp',       lang, xp=xp)}\n"
        f"{t('stats_answered', lang, total=total_q, pct=accuracy)}\n"
        f"{t('stats_rank',     lang, rank=rank)}"
    )


async def _safe_send(bot, chat_id: int, text: str, **kwargs) -> bool:
    for attempt in (1, 2):
        try:
            await bot.send_message(chat_id=chat_id, text=text, **kwargs)
            return True
        except Exception as e:
            logger.warning(f"send_message attempt {attempt} failed: {e}")
            if attempt == 1:
                await asyncio.sleep(1.0)
    return False


async def _send_question(chat_id: int, user_data: dict, bot) -> bool:
    session = user_data.get("session")
    if not session:
        return False

    questions = session["questions"]
    idx = session["current"]
    lang = get_lang(user_data)

    if idx >= len(questions):
        return await _send_final_score(chat_id, user_data, bot)

    q = questions[idx]
    quiz_id = session["quiz_id"]
    keyboard = build_question_keyboard(quiz_id, idx)

    md_text = format_question(q, idx + 1, len(questions), session["difficulty"], lang)
    sent = await _safe_send(bot, chat_id, md_text, parse_mode="Markdown", reply_markup=keyboard)

    if not sent:
        plain = format_plain_question(q, idx + 1, len(questions))
        sent = await _safe_send(bot, chat_id, plain, reply_markup=keyboard)

    if not sent:
        logger.error(
            f"FAILED to send Q{idx + 1}/{len(questions)} after all retries — "
            f"quiz session preserved for user_id={session.get('user_id')}"
        )
        await _safe_send(bot, chat_id, t("send_failed_alive", lang))
        return False

    return True


async def _send_final_score(chat_id: int, user_data: dict, bot) -> bool:
    session = user_data.get("session")
    if not session:
        return False

    lang = get_lang(user_data)
    score = session["score"]
    questions = session["questions"]
    total = len(questions)
    pct = round(score / total * 100) if total > 0 else 0
    diff = session["difficulty"]
    wrong_answers = session.get("wrong_answers", [])
    wrong_count = len(wrong_answers)

    if   pct >= 90: grade_key = "grade_outstanding"
    elif pct >= 80: grade_key = "grade_excellent"
    elif pct >= 70: grade_key = "grade_good"
    elif pct >= 60: grade_key = "grade_satisfactory"
    elif pct >= 40: grade_key = "grade_review"
    else:           grade_key = "grade_study"

    bar = progress_bar(total, total)

    msg = (
        f"{t('complete_title', lang)}

"
        f"`{bar}`

"
        f"{t('lbl_score', lang)}        `{score} / {total}`   ({pct}%)
"
        f"{t('lbl_grade', lang)}        {t(grade_key, lang)}
"
        f"{t('lbl_difficulty', lang)}   {diff_label(diff, lang)}"
    )
    if wrong_count > 0:
        plural = "" if wrong_count == 1 else ("s" if lang == "en" else "")
        msg += f"\n\n{t('wrong_count', lang, n=wrong_count, plural=plural)}"

    xp_earned = int(session.get("xp_earned", 0) or 0)
    streak = get_streak(user_data)
    if xp_earned > 0:
        msg += f"\n\n{t('complete_xp', lang, xp=xp_earned)}"
    if streak > 0:
        sep = "\n" if xp_earned > 0 else "\n\n"
        msg += f"{sep}{t('complete_streak', lang, days=streak)}"

    keyboard = build_post_quiz_keyboard(has_wrong=bool(wrong_answers), lang=lang)

    sent = await _safe_send(bot, chat_id, msg, parse_mode="Markdown", reply_markup=keyboard)
    if not sent:
        sent = await _safe_send(
            bot, chat_id,
            t("plain_final", lang, score=score, total=total, pct=pct),
            reply_markup=keyboard,
        )

    if not sent:
        logger.error(
            "_send_final_score: delivery failed — session preserved for user_id=%s",
            session.get("user_id"),
        )
        return False

    update_stats(user_data, score, total, diff)
    user_data.pop("session", None)
    user_data["review_pool"] = wrong_answers if wrong_answers else []

    if not user_data.get("first_quiz_prompt_done"):
        user_data["first_quiz_prompt_done"] = True
        try:
            await _safe_send(
                bot, chat_id, t("first_quiz_prompt", lang),
                parse_mode="Markdown",
                reply_markup=build_first_quiz_prompt_keyboard(lang),
            )
        except Exception as e:
            logger.warning("first_quiz_prompt delivery failed: %s", e)

    return True


async def send_review_question(chat_id: int, context: ContextTypes.DEFAULT_TYPE) -> None:
    lang = get_lang(context.user_data)
    review = context.user_data.get("review")
    if not review:
        await context.bot.send_message(chat_id=chat_id, text=t("review_no_active", lang))
        return

    pool = review["pool"]
    current = review["current"]

    if current >= len(pool):
        context.user_data.pop("review", None)
        await context.bot.send_message(chat_id=chat_id, text=t("review_complete", lang), parse_mode="Markdown")
        return

    entry = pool[current]
    text = format_review_question(entry, current + 1, len(pool), lang)
    keyboard = build_review_nav_keyboard(current, len(pool), lang)

    sent = await _safe_send(context.bot, chat_id, text, parse_mode="Markdown", reply_markup=keyboard)
    if not sent:
        q = entry["question"]
        correct = q["correct"].upper()
        opts = q["options"]
        plain = (
            f"{t('plain_review_header', lang, i=current + 1, n=len(pool))}\n\n"
            f"{q['question']}\n\n"
            f"{t('plain_answered', lang, chosen=entry['chosen'])}\n"
            f"{t('plain_correct_line', lang, letter=correct, opt=opts[correct])}\n\n"
            f"{q['explanation']}"
        )
        await _safe_send(context.bot, chat_id, plain, reply_markup=keyboard)


async def handle_review_callback(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    query = update.callback_query
    await query.answer()

    lang = get_lang(context.user_data)
    action = query.data.split(":")[1]
    chat_id = update.effective_chat.id

    if action == "start":
        pool = context.user_data.get("review_pool", [])
        if not pool:
            await query.edit_message_text(t("review_no_wrong_inline", lang))
            return
        context.user_data["review"] = {"pool": pool, "current": 0}
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await send_review_question(chat_id, context)

    elif action == "next":
        review = context.user_data.get("review")
        if not review:
            await query.edit_message_text(t("review_no_active", lang))
            return
        review["current"] += 1
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await send_review_question(chat_id, context)

    elif action == "exit":
        context.user_data.pop("review", None)
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await context.bot.send_message(chat_id=chat_id, text=t("review_ended", lang))

    elif action == "stats":
        text = format_stats_text(context.user_data)
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await context.bot.send_message(
            chat_id=chat_id, text=text, parse_mode="Markdown",
            reply_markup=build_stats_keyboard(lang),
        )

    elif action == "new":
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await context.bot.send_message(chat_id=chat_id, text=t("send_pdf_hint", lang))

async def handle_menu_callback(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    query = update.callback_query
    await query.answer()

    parts = query.data.split(":")
    if len(parts) != 2 or parts[1] not in ("leaderboard", "back", "change_nickname"):
        return
    action = parts[1]
    lang = get_lang(context.user_data)

    _capture_display_name(context.user_data, update.effective_user)
    caller_uid = update.effective_user.id if update.effective_user else None

    if action == "change_nickname":
        context.user_data["awaiting_nickname"] = True
        logger.info("handle_menu_callback: change_nickname armed user_id=%s", caller_uid)
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        await context.bot.send_message(
            chat_id=update.effective_chat.id,
            text=t("nickname_ask", lang),
        )
        return

    if action == "back":
        kb = build_mode_keyboard(lang, context.user_data, caller_uid)
        try:
            await query.edit_message_text(
                t("choose_mode", lang),
                parse_mode="Markdown",
                reply_markup=kb,
            )
        except Exception:
            await context.bot.send_message(
                chat_id=update.effective_chat.id,
                text=t("choose_mode", lang),
                parse_mode="Markdown",
                reply_markup=kb,
            )
        return

    all_users = context.application.user_data or {}
    rows: list[tuple[int, str]] = []
    for uid, ud in all_users.items():
        if not isinstance(ud, dict):
            continue
        raw_xp = ud.get("xp", 0)
        try:
            xp = max(0, int(raw_xp or 0))
        except (TypeError, ValueError):
            xp = 0
        if xp <= 0:
            continue
        nick        = (ud.get("nickname") or "").strip()
        stored_name = (ud.get("display_name") or "").strip()
        stored_user = (ud.get("username") or "").strip()
        name = nick or stored_name or stored_user
        if not name and uid == caller_uid and update.effective_user:
            live_display, live_username = _best_display_name(update.effective_user)
            name = live_display or live_username
        if not name:
            name = t("leaderboard_anon", lang)
        rows.append((xp, name))
    rows.sort(key=lambda r: r[0], reverse=True)
    top = rows[:10]

    logger.info(
        "handle_menu_callback: leaderboard size=%d top_xp=%s user_id=%s",
        len(top), top[0][0] if top else 0,
        update.effective_user.id if update.effective_user else "unknown",
    )

    if not top:
        text = t("leaderboard_empty", lang)
    else:
        lines = [t("leaderboard_title", lang), ""]
        for i, (xp, name) in enumerate(top, start=1):
            lines.append(f"{i}. {name} — {xp} XP")
        text = "\n".join(lines)

    try:
        await query.edit_message_text(
            text, reply_markup=build_leaderboard_keyboard(lang),
        )
    except Exception:
        await context.bot.send_message(
            chat_id=update.effective_chat.id, text=text,
            reply_markup=build_leaderboard_keyboard(lang),
        )

def _count_unique_users_sync(dsn: str) -> int:
    conn = psycopg2.connect(dsn)
    try:
        with conn.cursor() as cur:
            cur.execute(
                "SELECT COUNT(DISTINCT key) FROM bot_persistence WHERE store = %s",
                ("user_data",),
            )
            row = cur.fetchone()
            return int(row[0]) if row and row[0] is not None else 0
    finally:
        conn.close()

async def _start_detective_session(
    update: Update,
    context: ContextTypes.DEFAULT_TYPE,
    msg,
    lang: str,
    source_text: str,
    source_label: str,
) -> None:
    chat_id = update.effective_chat.id
    user_id = (update.effective_user.id if update.effective_user else 0)

    try:
        await context.bot.send_chat_action(chat_id=chat_id, action="typing")
    except Exception:
        pass

    try:
        async with _API_SEMAPHORE:
            opening = await asyncio.to_thread(generate_detective_opening, source_text)
    except Exception as e:
        logger.error(
            "detective_open FAILED user_id=%s source=%s: %s",
            user_id, source_label, e, exc_info=True,
        )
        try:
            await msg.edit_text(t("detective_error", lang, err=str(e)), parse_mode="Markdown")
        except Exception:
            await context.bot.send_message(
                chat_id=chat_id,
                text=t("detective_error", lang, err=str(e)),
                parse_mode="Markdown",
            )
        return

    context.user_data["detective"] = {
        "context":   source_text[:8_000],
        "history":   [{"role": "assistant", "content": opening}],
        "round":     0,
        "max_rounds": DETECTIVE_MAX_ROUNDS,
        "finished":  False,
        "busy":      False,
        "source":    source_label,
    }

    logger.info(
        "detective_open OK user_id=%s source=%s opening_len=%d",
        user_id, source_label, len(opening),
    )

    try:
        await msg.delete()
    except Exception:
        pass

    await context.bot.send_message(chat_id=chat_id, text=opening)


async def _handle_detective_move(
    update: Update,
    context: ContextTypes.DEFAULT_TYPE,
    detective: dict,
) -> None:
    lang = get_lang(context.user_data)
    chat_id = update.effective_chat.id
    user_id = (update.effective_user.id if update.effective_user else 0)
    move = (update.message.text or "").strip()

    if not move:
        return

    if detective.get("busy"):
        await update.message.reply_text(t("detective_busy", lang))
        return

    max_rounds = int(detective.get("max_rounds", DETECTIVE_MAX_ROUNDS))
    next_round = int(detective.get("round", 0)) + 1
    history = list(detective.get("history") or [])
    context_text = detective.get("context", "")

    history.append({"role": "user", "content": move})
    detective["history"] = history
    detective["busy"] = True

    try:
        await context.bot.send_chat_action(chat_id=chat_id, action="typing")
    except Exception:
        pass

    placeholder = None
    try:
        placeholder = await update.message.reply_text(
            t("detective_thinking", lang), parse_mode="Markdown"
        )
    except Exception:
        placeholder = None

    try:
        async with _API_SEMAPHORE:
            reply = await asyncio.to_thread(
                generate_detective_turn,
                context_text, history, next_round, max_rounds,
            )
    except Exception as e:
        logger.error(
            "detective_turn FAILED user_id=%s round=%d/%d: %s",
            user_id, next_round, max_rounds, e, exc_info=True,
        )
        detective["busy"] = False
        if history and history[-1].get("role") == "user":
            history.pop()
            detective["history"] = history
        err_text = t("detective_error", lang, err=str(e))
        if placeholder is not None:
            try:
                await placeholder.edit_text(err_text, parse_mode="Markdown")
                return
            except Exception:
                pass
        await context.bot.send_message(chat_id=chat_id, text=err_text, parse_mode="Markdown")
        return

    history.append({"role": "assistant", "content": reply})
    detective["history"] = history
    detective["round"] = next_round
    detective["busy"] = False

    is_final = next_round >= max_rounds
    header_key = "detective_final_header" if is_final else "detective_round_header"
    header = t(header_key, lang, round=next_round, max=max_rounds)

    if placeholder is not None:
        try:
            await placeholder.edit_text(header, parse_mode="Markdown")
        except Exception:
            pass
    await context.bot.send_message(chat_id=chat_id, text=reply)

    if is_final:
        detective["finished"] = True
        context.user_data.pop("detective", None)
        context.user_data.pop("pending_mode", None)
        await context.bot.send_message(
            chat_id=chat_id, text=t("detective_finished", lang),
        )

    logger.info(
        "detective_turn OK user_id=%s round=%d/%d final=%s history_len=%d",
        user_id, next_round, max_rounds, is_final, len(history),
    )

async def handle_file(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    _msg = update.message
    logger.info(
        "handle_file: ENTERED user_id=%s has_photo=%s has_document=%s "
        "doc_mime=%s doc_name=%s",
        update.effective_user.id if update.effective_user else "unknown",
        bool(_msg.photo) if _msg else "?",
        bool(_msg.document) if _msg else "?",
        (_msg.document.mime_type if _msg and _msg.document else ""),
        (_msg.document.file_name if _msg and _msg.document else ""),
    )

    lang = get_lang(context.user_data)
    pending_mode = context.user_data.get("pending_mode", "quiz")

    for key in _QUIZ_STATE_KEYS:
        context.user_data.pop(key, None)

    context.user_data["user_id"] = update.effective_user.id if update.effective_user else 0

    message   = update.message
    is_photo  = bool(message.photo)
    document  = message.document

    if is_photo:
        file_type = "image"
        img_mime  = "image/jpeg"
    elif document:
        mime = (document.mime_type or "").lower()
        name = (document.file_name or "").lower()
        if mime == "application/pdf" or name.endswith(".pdf"):
            file_type = "pdf"
            img_mime  = None
        elif mime in ("image/jpeg", "image/jpg") or name.endswith((".jpg", ".jpeg")):
            file_type = "image"
            img_mime  = "image/jpeg"
        elif mime == "image/png" or name.endswith(".png"):
            file_type = "image"
            img_mime  = "image/png"
        elif (mime == "application/vnd.openxmlformats-officedocument.wordprocessingml.document"
              or name.endswith(".docx")):
            file_type = "docx"
            img_mime  = None
        elif mime == "text/plain" or name.endswith(".txt"):
            file_type = "txt"
            img_mime  = None
        else:
            await message.reply_text(t("file_unsupported", lang))
            return
    else:
        await message.reply_text(t("file_unsupported", lang))
        return

    msg = await message.reply_text(t("file_received", lang))

    try:
        if is_photo:
            tg_file   = await message.photo[-1].get_file()
            raw_bytes = bytes(await tg_file.download_as_bytearray())
        else:
            tg_file   = await document.get_file()
            raw_bytes = bytes(await tg_file.download_as_bytearray())

        logger.info(
            "handle_file: downloaded %d bytes file_type=%s user_id=%s",
            len(raw_bytes), file_type, context.user_data.get("user_id"),
        )

        file_md5 = await asyncio.to_thread(_md5_bytes, raw_bytes)
        context.user_data["file_md5"] = file_md5

        try:
            await context.bot.send_chat_action(
                chat_id=update.effective_chat.id, action="typing"
            )
        except Exception:
            pass

        extract_anim = None
        if file_type != "txt":
            extract_anim = asyncio.create_task(animate_loading_message(msg))
        try:
            if file_type == "pdf":
                async with _API_SEMAPHORE:
                    text = await extract_text_from_pdf_async(raw_bytes)
            elif file_type == "image":
                async with _API_SEMAPHORE:
                    text = await asyncio.to_thread(extract_text_from_image, raw_bytes, img_mime)
            elif file_type == "docx":
                async with _API_SEMAPHORE:
                    text = await asyncio.to_thread(extract_text_from_docx, raw_bytes)
            else:
                text = extract_text_from_txt(raw_bytes)
        finally:
            await _cancel_loading_animation(extract_anim)

        if not text.strip():
            await msg.edit_text(t("file_no_text", lang))
            return

        if pending_mode == "summary":
            style = context.user_data.get("pending_summary_style", "disease")
            await msg.edit_text(t("summary_generating", lang), parse_mode="Markdown")

            summary_key = (file_md5, "summary", style)
            summary = _cache_get(summary_key)
            if summary is not None:
                logger.info(
                    "handle_file: summary CACHE HIT md5=%s style=%s user_id=%s",
                    file_md5[:10], style, context.user_data.get("user_id"),
                )
            else:
                summary_anim = asyncio.create_task(animate_loading_message(msg))
                try:
                    async with _API_SEMAPHORE:
                        summary = await asyncio.to_thread(generate_summary, text, style)
                finally:
                    await _cancel_loading_animation(summary_anim)
                _cache_put(summary_key, summary)
                logger.info(
                    "handle_file: summary CACHE MISS — stored md5=%s style=%s user_id=%s",
                    file_md5[:10], style, context.user_data.get("user_id"),
                )

            try:
                await msg.delete()
            except Exception:
                pass

            header = t("summary_header", lang)
            full   = header + summary

            async def _send_summary(txt: str) -> None:
                try:
                    await message.reply_text(txt, parse_mode="Markdown")
                except Exception:
                    await message.reply_text(txt)

            if len(full) <= 4000:
                await _send_summary(full)
            else:
                await _send_summary(header.strip())
                for chunk in _split_summary_chunks(summary, max_len=3800):
                    await _send_summary(chunk)

            logger.info(
                "handle_file: summary delivered user_id=%s file_type=%s",
                context.user_data.get("user_id"), file_type,
            )

        elif pending_mode == "detective":
            await _start_detective_session(
                update=update,
                context=context,
                msg=msg,
                lang=lang,
                source_text=text,
                source_label=f"file:{file_type}",
            )

        else:
            setup_id = str(uuid.uuid4())[:8]
            context.user_data["setup_id"]    = setup_id
            context.user_data["pending_text"] = text

            logger.info(
                "handle_file: text extracted ok, setup_id=%s user_id=%s file_type=%s",
                setup_id, context.user_data.get("user_id"), file_type,
            )

            await msg.edit_text(
                "كيف تفضل أن يكون نمط الأسئلة؟ / Choose the question style:",
                reply_markup=build_question_style_keyboard(setup_id),
            )

    except Exception as e:
        logger.error("handle_file error: %s", e, exc_info=True)
        context.user_data.pop("pending_text", None)
        context.user_data.pop("setup_id", None)
        await msg.edit_text(t("pdf_failed", lang, err=e))

async def handle_difficulty_selection(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    query = update.callback_query
    await query.answer()
    lang    = get_lang(context.user_data)
    user_id = update.effective_user.id if update.effective_user else "unknown"

    parts = query.data.split(":")
    if len(parts) != 3:
        logger.info(
            "handle_difficulty_selection: malformed cb=%r user_id=%s — ignoring",
            query.data, user_id,
        )
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        return

    _, cb_setup_id, difficulty = parts
    session_setup_id = context.user_data.get("setup_id")
    has_pending_text = bool(context.user_data.get("pending_text"))

    logger.info(
        "handle_difficulty_selection: user_id=%s cb=%r cb_setup_id=%s "
        "session_setup_id=%s pending_text=%s difficulty=%s",
        user_id, query.data, cb_setup_id, session_setup_id,
        "yes" if has_pending_text else "no", difficulty,
    )

    if session_setup_id is not None and cb_setup_id != session_setup_id:
        logger.info(
            "handle_difficulty_selection: stale setup_id cb=%s session=%s — discarding (user_id=%s)",
            cb_setup_id, session_setup_id, user_id,
        )
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        return

    if session_setup_id is None and not has_pending_text:
        logger.info(
            "handle_difficulty_selection: no setup_id and no pending_text — "
            "stale button, discarding (user_id=%s)", user_id,
        )
        try:
            await query.edit_message_reply_markup(reply_markup=None)
        except Exception:
            pass
        return

    if session_setup_id is None and has_pending_text:
        logger.info(
            "handle_difficulty_selection: legacy fallback (no saved setup_id, "
            "pending_text present) — accepting (user_id=%s)", user_id,
        )

    if session_setup_id is not None and not has_pending_text:
        logger.warning(
            "handle_difficulty_selection: matching setup_id but no pending_text user_id=%s", user_id,
        )
        try:
            await query.edit_message_text(t("stale_button", lang))
        except Exception:
            pass
        return

    if difficulty not in ("easy", "medium", "hard", "nightmare"):
        logger.warning(
            "handle_difficulty_selection: unknown difficulty=%r user_id=%s — ignoring",
            difficulty, user_id,
        )
        return

    if difficulty == "nightmare":
        result = await spend_unlock(
            context.application, context.user_data, user_id,
            XP_UNLOCK_COST, "nightmare",
        )
        if result == "insufficient":
            current = get_xp(context.user_data)
            logger.info(
                "handle_difficulty_selection: nightmare BLOCKED — insufficient XP user_id=%s have=%d need=%d",
                user_id, current, XP_UNLOCK_COST,
            )
            try:
                await query.answer(text=t("xp_locked", lang), show_alert=True)
            except Exception:
                await query.message.reply_text(t("xp_locked", lang))
            return
        if result == "bypass":
            logger.info("handle_difficulty_selection: nightmare ADMIN BYPASS user_id=%s", user_id)
        elif result == "duplicate":
            logger.info("handle_difficulty_selection: nightmare duplicate tap deduped user_id=%s", user_id)
        elif result == "ok":
            logger.info(
                "handle_difficulty_selection: nightmare UNLOCKED — spent %d XP user_id=%s new_balance=%d",
                XP_UNLOCK_COST, user_id, get_xp(context.user_data),
            )
            try:
                await query.message.reply_text(
                    t("xp_spent", lang, xp=get_xp(context.user_data)),
                )
            except Exception:
                pass

    context.user_data["pending_difficulty"] = difficulty
    logger.info(
        "handle_difficulty_selection: saved pending_difficulty=%s user_id=%s",
        difficulty, user_id,
    )

    try:
        await query.edit_message_text(
            t("diff_selected", lang, diff=diff_label(difficulty, lang)),
            parse_mode="Markdown",
            reply_markup=build_count_keyboard(),
        )
    except Exception as e:
        logger.error(
            "handle_difficulty_selection: edit_message_text failed user_id=%s: %s",
            user_id, e, exc_info=True,
        )
        await query.message.reply_text(
            t("diff_selected", lang, diff=diff_label(difficulty, lang)),
            parse_mode="Markdown",
            reply_markup=build_count_keyboard(),
        )

async def handle_count_selection(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    query = update.callback_query
    await query.answer()
    lang    = get_lang(context.user_data)
    user_id = update.effective_user.id if update.effective_user else "unknown"

    text       = context.user_data.get("pending_text")
    difficulty = context.user_data.get("pending_difficulty")

    logger.info(
        "handle_count_selection RECEIVED: cb=%r user_id=%s pending_text=%s "
        "pending_difficulty=%s",
        query.data, user_id, "yes" if text else "no", difficulty,
    )

    if not text or not difficulty:
        logger.info(
            "handle_count_selection: missing setup state — stale button (user_id=%s)",
            user_id,
        )
        try:
            await query.edit_message_text(t("stale_button", lang))
        except Exception:
            pass
        return

    parts = query.data.split(":")
    if len(parts) != 2:
        logger.warning("handle_count_selection: malformed cb=%r user_id=%s — ignoring", query.data, user_id)
        return
    try:
        count = int(parts[1])
    except ValueError:
        logger.warning("handle_count_selection: non-numeric count cb=%r user_id=%s — ignoring", query.data, user_id)
        return
    if count not in (5, 10, 20, 35, 50):
        logger.warning("handle_count_selection: out-of-range count=%s user_id=%s — ignoring", count, user_id)
        return

    context.user_data["pending_count"] = count
    logger.info(
        "handle_count_selection: difficulty=%s count=%d user_id=%s — generating",
        difficulty, count, user_id,
    )

    await query.edit_message_text(
        t("generating", lang, count=count, diff=diff_label(difficulty, lang)),
        parse_mode="Markdown",
    )

    try:
        await context.bot.send_chat_action(
            chat_id=update.effective_chat.id, action="typing"
        )
    except Exception:
        pass

    try:
        file_md5 = context.user_data.get("file_md5")
        style    = context.user_data.get("pending_style")
        mcq_key  = (file_md5, "quiz", difficulty, count, style) if file_md5 else None
        questions = _cache_get(mcq_key) if mcq_key else None

        if questions is not None:
            for q in questions:
                shuffle_question_options(q)
            logger.info(
                "handle_count_selection: MCQ CACHE HIT md5=%s diff=%s style=%s count=%d user_id=%s",
                file_md5[:10], difficulty, style or "default", count, user_id,
            )
        else:
            mcq_anim = asyncio.create_task(animate_loading_message(query.message))
            try:
                async with _API_SEMAPHORE:
                    questions = await asyncio.to_thread(
                        generate_mcqs, text, count, difficulty, style
                    )
            finally:
                await _cancel_loading_animation(mcq_anim)
            if questions and mcq_key:
                _cache_put(mcq_key, questions)
                logger.info(
                    "handle_count_selection: MCQ CACHE MISS — stored md5=%s diff=%s style=%s count=%d user_id=%s",
                    file_md5[:10] if file_md5 else "n/a",
                    difficulty, style or "default", count, user_id,
                )

        if not questions:
            logger.warning("handle_count_selection: generator returned 0 questions user_id=%s", user_id)
            context.user_data.pop("pending_count", None)
            try:
                await query.edit_message_text(
                    t("no_questions", lang),
                    reply_markup=build_count_keyboard(),
                )
            except Exception as e:
                logger.warning("handle_count_selection: edit on no_questions failed user_id=%s: %s", user_id, e, exc_info=True)
            return

        sess_user_id = context.user_data.get("user_id", update.effective_user.id)
        quiz_id      = generate_quiz_id()

        context.user_data["session"] = {
            "quiz_id":       quiz_id,
            "user_id":       sess_user_id,
            "questions":     questions,
            "current":       0,
            "score":         0,
            "difficulty":    difficulty,
            "wrong_answers": [],
            "xp_earned":     0,
        }

        new_streak = update_streak(context.user_data)
        logger.info("handle_count_selection: streak now=%d user_id=%s", new_streak, user_id)

        logger.info(
            "handle_count_selection: built session quiz_id=%s questions=%d difficulty=%s user_id=%s",
            quiz_id, len(questions), difficulty, user_id,
        )

        await query.edit_message_text(
            t("ready", lang, count=len(questions), diff=diff_label(difficulty, lang)),
            parse_mode="Markdown",
        )

        delivered = await _send_question(
            update.effective_chat.id,
            context.user_data,
            context.bot,
        )

        if delivered:
            context.user_data.pop("pending_text", None)
            context.user_data.pop("pending_difficulty", None)
            context.user_data.pop("pending_style", None)
            context.user_data.pop("selected_style", None)
            context.user_data.pop("pending_count", None)
            context.user_data.pop("setup_id", None)
            logger.info("handle_count_selection: quiz started quiz_id=%s user_id=%s", quiz_id, user_id)
        else:
            logger.error("handle_count_selection: Q1 delivery failed, setup state preserved quiz_id=%s user_id=%s", quiz_id, user_id)

    except json.JSONDecodeError as e:
        logger.error("JSON parse error user_id=%s: %s", user_id, e, exc_info=True)
        context.user_data.pop("pending_count", None)
        try:
            await query.edit_message_text(
                t("json_error", lang),
                reply_markup=build_count_keyboard(),
            )
        except Exception as ee:
            logger.warning("handle_count_selection: edit on json_error failed user_id=%s: %s", user_id, ee, exc_info=True)
    except Exception as e:
        logger.error("MCQ generation error user_id=%s: %s", user_id, e, exc_info=True)
        context.user_data.pop("pending_count", None)
        try:
            await query.edit_message_text(
                t("mcq_error", lang, err=e),
                reply_markup=build_count_keyboard(),
            )
        except Exception as ee:
            logger.warning("handle_count_selection: edit on mcq_error failed user_id=%s: %s", user_id, ee, exc_info=True)

_ANSWER_LOCKS: Dict[int, asyncio.Lock] = {}
_LAST_CLICK_TIME: Dict[int, float] = {}
_ANSWER_DEBOUNCE_SECONDS = 0.7


def _get_answer_lock(user_id: int) -> asyncio.Lock:
    lock = _ANSWER_LOCKS.get(user_id)
    if lock is None:
        lock = asyncio.Lock()
        _ANSWER_LOCKS[user_id] = lock
    return lock


async def handle_answer(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    query = update.callback_query
    await query.answer()

    parts = query.data.split(":")
    if len(parts) != 4:
        return
    _, cb_quiz_id, cb_idx_str, chosen = parts
    try:
        cb_idx = int(cb_idx_str)
    except ValueError:
        return

    chat_id  = update.effective_chat.id
    user_id  = update.effective_user.id if update.effective_user else "unknown"
    lang     = get_lang(context.user_data)
    _lock_uid = update.effective_user.id if update.effective_user else 0

    now = time.monotonic()
    last = _LAST_CLICK_TIME.get(_lock_uid)
    if last is not None and (now - last) < _ANSWER_DEBOUNCE_SECONDS:
        logger.info("handle_answer: DEBOUNCED tap (Δ=%.3fs < %.2fs) user_id=%s cb=%r", now - last, _ANSWER_DEBOUNCE_SECONDS, user_id, query.data)
        return
    _LAST_CLICK_TIME[_lock_uid] = now

    _snapshot = context.user_data.get("session")
    if _snapshot is None:
        logger.info("handle_answer: pre-lock fast-fail (no session) user_id=%s", user_id)
        return
    _snap_quiz_id = _snapshot.get("quiz_id")
    _snap_current = _snapshot.get("current")
    if _snap_quiz_id != cb_quiz_id:
        logger.info("handle_answer: pre-lock fast-fail (stale quiz_id cb=%s session=%s) user_id=%s", cb_quiz_id, _snap_quiz_id, user_id)
        return
    if isinstance(_snap_current, int) and cb_idx < _snap_current:
        logger.info("handle_answer: pre-lock fast-fail (cb_idx=%d < current=%d) user_id=%s", cb_idx, _snap_current, user_id)
        return

    async with _get_answer_lock(_lock_uid):
        session = context.user_data.get("session")

        logger.info("handle_answer RECEIVED: user_id=%s cb_quiz_id=%s cb_idx=%s chosen=%s cb=%r", user_id, cb_quiz_id, cb_idx, chosen, query.data)

        if not session:
            try:
                await query.edit_message_reply_markup(reply_markup=None)
            except Exception:
                pass
            logger.info("handle_answer: no session — ignoring (user_id=%s)", user_id)
            return

        if cb_quiz_id != session.get("quiz_id"):
            try:
                await query.edit_message_reply_markup(reply_markup=None)
            except Exception:
                pass
            logger.info("handle_answer: stale quiz_id cb=%s session=%s — ignoring (user_id=%s)", cb_quiz_id, session.get("quiz_id"), user_id)
            return

        if cb_idx != session.get("current"):
            try:
                await query.edit_message_reply_markup(reply_markup=None)
            except Exception:
                pass
            logger.info("handle_answer: cb_idx=%s != current=%s — duplicate/stale tap ignored (user_id=%s)", cb_idx, session.get("current"), user_id)
            return

        questions  = session["questions"]
        idx        = session["current"]
        total      = len(questions)
        quiz_id    = session.get("quiz_id")
        difficulty = session["difficulty"]
    
        logger.info("handle_answer PROCESSING: user_id=%s quiz_id=%s Q%d/%d chosen=%s", user_id, quiz_id, idx + 1, total, chosen)
    
        q       = questions[idx]
        correct = q["correct"].upper()
        opts    = q["options"]
    
        is_correct = chosen == correct
        if is_correct:
            session["score"] += 1
            result_line = t("correct", lang)
        else:
            result_line = t("wrong", lang, chosen=chosen)
            session.setdefault("wrong_answers", []).append({"question": q, "chosen": chosen})

        try:
            xp_delta = award_xp(context.user_data, difficulty, is_correct)
            if xp_delta > 0:
                session["xp_earned"] = int(session.get("xp_earned", 0) or 0) + xp_delta
            logger.info("handle_answer XP: user_id=%s quiz_id=%s Q%d/%d correct=%s delta=%+d total=%d", user_id, quiz_id, idx + 1, total, is_correct, xp_delta, get_xp(context.user_data))
        except Exception:
            logger.exception("handle_answer: award_xp failed; ignoring")
    
        session["current"] += 1
        next_idx = session["current"]
    
        logger.info("handle_answer ADVANCED: user_id=%s quiz_id=%s current=%d/%d correct=%s is_correct=%s", user_id, quiz_id, next_idx, total, correct, is_correct)
    
        try:
            full_text: str | None = None
            try:
                feedback = (
                    f"{result_line}\n"
                    f"{t('answer_label', lang)} *{correct}.* "
                    f"{escape_md(opts.get(correct, correct))}\n\n"
                    f"{t('explanation', lang)} {escape_md(q.get('explanation', ''))}"
                )
                original_text = format_question(q, idx + 1, total, difficulty, lang)
                full_text = original_text + "\n\n" + feedback
            except Exception as e:
                logger.error("handle_answer: feedback construction failed user_id=%s quiz_id=%s Q%d: %s", user_id, quiz_id, idx + 1, e, exc_info=True)
    
            edit_ok = False
            if full_text is not None:
                try:
                    await query.edit_message_text(text=full_text, parse_mode="Markdown", reply_markup=None)
                    edit_ok = True
                except Exception as e:
                    logger.error("handle_answer: edit_message_text failed user_id=%s quiz_id=%s Q%d: %s", user_id, quiz_id, idx + 1, e, exc_info=True)
    
            if not edit_ok:
                try:
                    await query.edit_message_reply_markup(reply_markup=None)
                except Exception as e:
                    logger.warning("handle_answer: edit_message_reply_markup fallback failed user_id=%s quiz_id=%s Q%d: %s", user_id, quiz_id, idx + 1, e, exc_info=True)
                try:
                    head  = t("plain_correct", lang) if is_correct else t("plain_wrong", lang, chosen=chosen)
                    plain = (
                        f"{head}\n\n"
                        f"{t('plain_correct_line', lang, letter=correct, opt=opts.get(correct, correct))}\n\n"
                        f"{t('plain_explanation', lang, text=q.get('explanation', ''))}"
                    )
                    await _safe_send(context.bot, chat_id, plain)
                except Exception as e:
                    logger.error("handle_answer: plain feedback send failed user_id=%s quiz_id=%s Q%d: %s", user_id, quiz_id, idx + 1, e, exc_info=True)
    
            await asyncio.sleep(0.5)
    
            delivered = False
            last_err: Exception | None = None
            for attempt in (1, 2):
                try:
                    if next_idx >= total:
                        ok = await _send_final_score(chat_id, context.user_data, context.bot)
                    else:
                        ok = await _send_question(chat_id, context.user_data, context.bot)
                    if ok:
                        delivered = True
                        logger.info("handle_answer DELIVERED: user_id=%s quiz_id=%s Q%d done attempt=%d next=%s", user_id, quiz_id, idx + 1, attempt, "final" if next_idx >= total else f"Q{next_idx + 1}")
                        break
                    logger.error("handle_answer: delivery attempt %d returned False: user_id=%s quiz_id=%s Q%d -> %s", attempt, user_id, quiz_id, idx + 1, "final" if next_idx >= total else f"Q{next_idx + 1}")
                    last_err = RuntimeError("send returned False")
                    if attempt == 1:
                        await asyncio.sleep(1.5)
                except Exception as e:
                    last_err = e
                    logger.error("handle_answer: delivery attempt %d raised: user_id=%s quiz_id=%s Q%d -> %s: %s", attempt, user_id, quiz_id, idx + 1, "final" if next_idx >= total else f"Q{next_idx + 1}", e, exc_info=True)
                    if attempt == 1:
                        await asyncio.sleep(1.5)
    
            if not delivered:
                logger.error("handle_answer: ALL delivery attempts failed: user_id=%s quiz_id=%s Q%d last_err=%s — notifying user", user_id, quiz_id, idx + 1, last_err)
                await _safe_send(context.bot, chat_id, t("answer_error_recovery", lang))
    
        except Exception as _top_err:
            logger.error("handle_answer: unexpected top-level exception after current advance — stripping buttons and recovering: user_id=%s quiz_id=%s Q%d: %s", user_id, quiz_id, idx + 1, _top_err, exc_info=True)
            try:
                await query.edit_message_reply_markup(reply_markup=None)
            except Exception:
                pass
            await _safe_send(context.bot, chat_id, t("answer_error_recovery", lang))
            try:
                await _send_question(chat_id, context.user_data, context.bot)
            except Exception as _recover_err:
                logger.error("handle_answer: recovery delivery also failed user_id=%s quiz_id=%s: %s", user_id, quiz_id, _recover_err, exc_info=True)
                await _safe_send(context.bot, chat_id, t("answer_error_recovery", lang))

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    lang = get_lang(context.user_data)

    if context.user_data.get("awaiting_nickname"):
        raw = update.message.text or ""
        clean, err = _validate_nickname(raw)
        if err is not None:
            await update.message.reply_text(t(err, lang))
            return
        context.user_data["nickname"] = clean
        context.user_data.pop("awaiting_nickname", None)
        safe_clean = clean.replace("_", r"\_")
        try:
            await context.application.update_persistence()
        except Exception as e:
            logger.warning("nickname persist flush failed: %s", e)
        logger.info("handle_message: nickname updated user_id=%s len=%d", update.effective_user.id if update.effective_user else "unknown", len(clean))
        await update.message.reply_text(
            t("nickname_saved", lang, name=safe_clean),
            parse_mode="Markdown",
        )
        user = update.effective_user
        await update.message.reply_text(
            t("choose_mode", lang),
            parse_mode="Markdown",
            reply_markup=build_mode_keyboard(
                lang, context.user_data, user.id if user else None,
            ),
        )
        return

    detective = context.user_data.get("detective")
    pending_mode = context.user_data.get("pending_mode")

    if detective and isinstance(detective, dict) and not detective.get("finished"):
        await _handle_detective_move(update, context, detective)
        return

    if pending_mode == "detective" and not detective:
        text = (update.message.text or "").strip()
        if not text:
            await update.message.reply_text(t("detective_no_input", lang))
            return
        msg = await update.message.reply_text(
            t("detective_opening", lang), parse_mode="Markdown"
        )
        await _start_detective_session(
            update=update,
            context=context,
            msg=msg,
            lang=lang,
            source_text=text,
            source_label="topic",
        )
        return

    if context.user_data.get("session"):
        await update.message.reply_text(t("quiz_in_progress", lang))
    else:
        await update.message.reply_text(t("send_pdf_hint", lang))

        if affected_user_ids:
            try:
                application.mark_data_for_update_persistence(user_ids={int(uid) for uid in affected_user_ids})
                await application.update_persistence()
                logger.info("Stripped legacy 'time_limit' from %d persisted session(s)", len(affected_user_ids))
            except Exception:
                logger.exception("Failed to flush stripped 'time_limit' to persistence")
    except Exception:
        logger.exception("Failed to strip legacy 'time_limit' from persisted sessions")


async def _error_handler(update: object, context: ContextTypes.DEFAULT_TYPE) -> None:
    logger.error("Unhandled exception", exc_info=context.error)
    if isinstance(update, Update) and update.effective_message:
        lang = "en"
        try:
            if context.user_data:
                lang = get_lang(context.user_data)
        except Exception:
            pass
        try:
            await update.effective_message.reply_text(t("unhandled_error", lang))
        except Exception:
            pass


def main() -> None:
    dsn = os.environ["DATABASE_URL"]

    pkl_path = "quiz_sessions.pkl"
    if os.path.exists(pkl_path):
        _migrate_pickle_to_postgres(pkl_path, dsn)

    persistence = PostgresPersistence(dsn=dsn)
    app = (
        ApplicationBuilder()
        .token(TOKEN)
        .persistence(persistence)
        .concurrent_updates(True)
        .post_init(_post_init)
        .build()
    )

    app.add_error_handler(_error_handler)

    app.add_handler(CommandHandler("start",    start))
    app.add_handler(CommandHandler("help",     help_command))
    app.add_handler(CommandHandler("stop",     stop_command))
    app.add_handler(CommandHandler("stats",    stats_command))
    app.add_handler(CommandHandler("review",   review_command))
    app.add_handler(CommandHandler("language", language_command))
    app.add_handler(CommandHandler("support",  support_command))
    app.add_handler(CommandHandler("testbot",  testbot_command))
    app.add_handler(CommandHandler("admin_stats", admin_stats_command))

    class _SupportedFileFilter(filters.MessageFilter):
        _MIMES = {
            "application/pdf",
            "image/jpeg", "image/jpg", "image/png",
            "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
            "text/plain",
        }
        _EXTS = (".pdf", ".jpg", ".jpeg", ".png", ".docx", ".txt")

        def filter(self, message) -> bool:
            doc = getattr(message, "document", None)
            if not doc:
                return False
            mime = (getattr(doc, "mime_type", "") or "").lower()
            name = (getattr(doc, "file_name", "") or "").lower()
            return mime in self._MIMES or name.endswith(self._EXTS)

    supported_file_filter = _SupportedFileFilter()
    app.add_handler(MessageHandler(filters.PHOTO,                                      handle_file))
    app.add_handler(MessageHandler(supported_file_filter,                              handle_file))
    app.add_handler(MessageHandler(filters.Document.ALL & ~supported_file_filter,      handle_non_pdf_document))

    app.add_handler(CallbackQueryHandler(handle_setlang_callback,     pattern=r"^setlang:"))
    app.add_handler(CallbackQueryHandler(handle_lang_callback,        pattern=r"^lang:"))
    app.add_handler(CallbackQueryHandler(handle_mode_callback,             pattern=r"^mode:"))
    app.add_handler(CallbackQueryHandler(handle_menu_callback,             pattern=r"^menu:"))
    app.add_handler(CallbackQueryHandler(handle_summary_style_callback,   pattern=r"^style:"))
    app.add_handler(CallbackQueryHandler(handle_setup_back_callback,      pattern=r"^setup:back$"))
    app.add_handler(CallbackQueryHandler(handle_qstyle_selection,     pattern=r"^qstyle:"))
    app.add_handler(CallbackQueryHandler(handle_difficulty_selection, pattern=r"^difficulty:"))
    app.add_handler(CallbackQueryHandler(handle_legacy_diff_callback, pattern=r"^diff:"))
    app.add_handler(CallbackQueryHandler(handle_count_selection,      pattern=r"^count:"))
    app.add_handler(CallbackQueryHandler(button_click,                pattern=r"^support_madar$"))
    app.add_handler(CallbackQueryHandler(handle_answer,               pattern=r"^answer:"))
    app.add_handler(CallbackQueryHandler(handle_review_callback,      pattern=r"^rv:"))

    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    async def _unhandled_update(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
        msg = update.message or update.edited_message or update.channel_post
        logger.warning(
            "UNHANDLED UPDATE: update_id=%s has_photo=%s has_document=%s "
            "has_text=%s effective_chat=%s",
            update.update_id,
            bool(msg.photo) if msg else "?",
            bool(msg.document) if msg else "?",
            bool(msg.text) if msg else "?",
            update.effective_chat.id if update.effective_chat else "?",
        )

    app.add_handler(MessageHandler(filters.ALL, _unhandled_update))

    logger.info("Bot is running...")
    app.run_polling(allowed_updates=Update.ALL_TYPES)


if __name__ == "__main__":
    main()
